# ========== HERRING BUSS (North Sea, c.1415) ==========
# Floating factories, salt-cure, extended range – marginal early fish preservation
# North Sea: age 2 | Coastal Europe: age 3 | General: age 4
pp_herring_buss_north_sea = {
	age = age_2_renaissance
	icon = safe_exploration_techniques_advance
	requires = rgo_size_advance_renaissance
	global_fish_output_modifier = 0.20
	ai_weight = { add = 50 }
	potential = {
		exists = capital
		original_capital ?= {
			OR = {
				region = region:scandinavian_region
				region = region:great_britain_region
				region = region:north_german_region
			}
		}
	}
}
