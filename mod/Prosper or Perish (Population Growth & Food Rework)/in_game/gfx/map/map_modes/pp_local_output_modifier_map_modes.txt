# Map modes for local good output modifiers (Geography category).
# Displays the aggregated local_<good>_output_modifier per location.
# Only goods modified in goods_output_modifiers.xlsx (vegetation, topography, climate, water).
# Red = -50%, Green = +150%. Values are additive percentages (0.1 = +10%).
# Geography `index` is a UI sub-group (vanilla uses only 0–3); PP local-output modes all use index = 2 (terrain/climate bucket). Values >3 are ignored in-game.

@factor_add = 0.5
@factor_divide = 2.0
@zoom_step_near = 2
@data_gradient_alpha_inside = 1
@data_gradient_alpha_outside = 1
@data_gradient_width = 0.25
@data_gradient_color_mult = 0.9
@data_edge_width = 0
@data_edge_sharpness = 0.01
@data_edge_alpha = 0
@data_edge_color_mult = 0
@data_before_lighting_blend = 0.5
@data_after_lighting_blend = 0.5

pp_local_fruit_output_modifier = {
	map_color = {
		if = {
			limit = { is_ownable = yes }
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = {
					value = modifier:local_fruit_output_modifier
					add = @factor_add
					divide = @factor_divide
				}
			}
		}
		else = { value = rgb { 128 128 128 } }  # grey/neutral for non-ownable (water, etc.)
	}
	legend_key = { desc = "MAPMODE_PP_LOCAL_FRUIT_OUTPUT_MODIFIER" color = define:NMapColors|MAP_COLOR_HIGH }
	tooltip_key = {
		if = {
			limit = { is_land = yes }
			value = MAPMODE_PP_LOCAL_FRUIT_OUTPUT_MODIFIER_TT_LAND_BREAKDOWN
		}
		else = {
			value = MAPMODE_PP_LOCAL_OUTPUT_TT_WATER
		}
	}
	small_map_names = location
	medium_map_names = location
	large_map_names = location
	small_tooltip_context = location
	medium_tooltip_context = location
	large_tooltip_context = location
	fill_in_impassable = yes
	enable_snow = no
	flatmap_behaviour = always
	use_fow = no
	category = geography
	index = 2
	allow_allocate_hotkey = no
	map_markers = { all = no }
	gradient_parameters = {
		zoom_step = @zoom_step_near
		gradient_alpha_inside = @data_gradient_alpha_inside
		gradient_alpha_outside = @data_gradient_alpha_outside
		gradient_width = @data_gradient_width
		gradient_color_mult = @data_gradient_color_mult
		edge_width = @data_edge_width
		edge_sharpness = @data_edge_sharpness
		edge_alpha = @data_edge_alpha
		edge_color_mult = @data_edge_color_mult
		before_lighting_blend = @data_before_lighting_blend
		after_lighting_blend = @data_after_lighting_blend
	}
}
