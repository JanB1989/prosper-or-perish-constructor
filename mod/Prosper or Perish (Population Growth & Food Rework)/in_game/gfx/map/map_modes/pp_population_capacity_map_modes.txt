# Map mode for population capacity per location.
# Green = high capacity, Red = low capacity. Uncolonized ownable land is included.
# Uses modifier:local_population_capacity (in 1k), scaled 0-300 for factor (0-1).

@zoom_step_near = 2
@data_gradient_alpha_inside = 1
@data_gradient_alpha_outside = 1
@data_gradient_width = 0.25
@data_gradient_color_mult = 0.9
@data_edge_width = 0
@data_edge_sharpness = 0.01
@data_edge_alpha = 0
@data_edge_color_mult = 0
@data_before_lighting_blend = 0.5
@data_after_lighting_blend = 0.5
@pp_population_capacity_map_cap = 300

# ~0.9% monthly local growth modifier (script fraction) — full gradient + stripe threshold (tune with balance).
@pp_population_growth_cap_stripe = 0.009

pp_population_capacity = {
	map_color = {
		if = {
			limit = { is_ownable = no }
			value = rgb { 128 128 128 }
		}
		else = {
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = {
					value = modifier:local_population_capacity
					divide = @pp_population_capacity_map_cap
					max = 1
					min = 0
				}
			}
		}
	}
	legend_key = { desc = "MAPMODE_PP_POPULATION_CAPACITY_LOW" color = define:NMapColors|MAP_COLOR_LOW }
	legend_key = { desc = "MAPMODE_PP_POPULATION_CAPACITY_HIGH" color = define:NMapColors|MAP_COLOR_HIGH }
	tooltip_key = {
		if = { limit = { is_land = yes } value = MAPMODE_PP_POPULATION_CAPACITY_TT_LAND }
		else = { value = MAPMODE_PP_POPULATION_CAPACITY_TT_WATER }
	}
	small_map_names = location
	medium_map_names = location
	large_map_names = location
	small_tooltip_context = location
	medium_tooltip_context = location
	large_tooltip_context = location
	fill_in_impassable = yes
	enable_snow = no
	flatmap_behaviour = always
	use_fow = no
	category = population
	index = 1
	map_markers = { all = no }
	gradient_parameters = {
		zoom_step = @zoom_step_near
		gradient_alpha_inside = @data_gradient_alpha_inside
		gradient_alpha_outside = @data_gradient_alpha_outside
		gradient_width = @data_gradient_width
		gradient_color_mult = @data_gradient_color_mult
		edge_width = @data_edge_width
		edge_sharpness = @data_edge_sharpness
		edge_alpha = @data_edge_alpha
		edge_color_mult = @data_edge_color_mult
		before_lighting_blend = @data_before_lighting_blend
		after_lighting_blend = @data_after_lighting_blend
	}
	color_refresh_counters = { LocationDevelopmentChanged LocationPopulationChanged }
	color_and_names_refresh_counters = { LocationOwnerChanged CountryStatus }
}

# Map mode for local population growth modifier per location.
# Heatmap: 0 to ~0.9% (0.009) uses the full gradient; negatives clamp to red.
# Stripes (secondary): local modifier at/above ~0.9%, same idea as vanilla Population stripes for over-cap.

pp_population_growth = {
	map_color = {
		if = {
			limit = { is_ownable = no }
			value = rgb { 128 128 128 }
		}
		else_if = {
			limit = { has_owner = yes }
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = {
					value = modifier:local_population_growth
					divide = @pp_population_growth_cap_stripe
					max = 1
					min = 0
				}
			}
		}
		else = {
			value = rgb { 128 128 128 }
		}
	}
	secondary_map_color = {
		if = {
			limit = {
				has_owner = yes
				modifier:local_population_growth >= @pp_population_growth_cap_stripe
			}
			value = rgb { 35 35 42 }
		}
	}
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH" color = define:NMapColors|MAP_COLOR_HIGH }
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH_STRIPE" color = rgb { 35 35 42 } }
	tooltip_key = {
		if = { limit = { is_land = yes } value = MAPMODE_PP_POPULATION_GROWTH_TT_LAND }
		else = { value = MAPMODE_PP_POPULATION_GROWTH_TT_WATER }
	}
	small_map_names = location
	medium_map_names = location
	large_map_names = location
	small_tooltip_context = location
	medium_tooltip_context = location
	large_tooltip_context = location
	fill_in_impassable = yes
	enable_snow = no
	flatmap_behaviour = always
	use_fow = no
	category = population
	index = 1
	map_markers = { all = no }
	gradient_parameters = {
		zoom_step = @zoom_step_near
		gradient_alpha_inside = @data_gradient_alpha_inside
		gradient_alpha_outside = @data_gradient_alpha_outside
		gradient_width = @data_gradient_width
		gradient_color_mult = @data_gradient_color_mult
		edge_width = @data_edge_width
		edge_sharpness = @data_edge_sharpness
		edge_alpha = @data_edge_alpha
		edge_color_mult = @data_edge_color_mult
		before_lighting_blend = @data_before_lighting_blend
		after_lighting_blend = @data_after_lighting_blend
	}
	color_refresh_counters = { LocationDevelopmentChanged LocationPopulationChanged }
	color_and_names_refresh_counters = { LocationOwnerChanged CountryStatus }
}
